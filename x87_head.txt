ROOT = _resolve_aios_root()
CORE_DIR = ROOT / "core"
RUNTIME_DIR = ROOT / "runtime"
BUS_DIR = RUNTIME_DIR / "x86_agent_bus"
MESSAGES_DIR = BUS_DIR / "messages"
REPORT_DIR = RUNTIME_DIR / "reports"
STATE_DIR = RUNTIME_DIR / "x87_worker_dispatcher"
STATE_FILE = STATE_DIR / "state.json"

ROUTE_NEXT = {
    "DEBUG": "PATCHER",
    "PATCHER": "VERIFY",
    "VERIFY": "REPORT",
}


def utc_stamp() -> str:
    return time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime())


def now_ms() -> int:
    return int(time.time() * 1000)


def safe_write_json(path: Path, data: Dict[str, Any]) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    tmp = path.with_suffix(path.suffix + ".tmp")
    tmp.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")
    tmp.replace(path)


def load_json(path: Path, default: Dict[str, Any]) -> Dict[str, Any]:
    if not path.exists():
        return default
    try:
        return json.loads(path.read_text(encoding="utf-8"))
    except Exception as exc:
        return {**default, "load_warning": f"Failed to load {path}: {exc}"}


def load_x86_bus_class():
    bus_file = CORE_DIR / "x86_agent_bus.py"
    if not bus_file.exists():
        raise FileNotFoundError(f"X86 dependency missing: {bus_file}")

    module_name = "x86_agent_bus_runtime"
    spec = importlib.util.spec_from_file_location(module_name, str(bus_file))
    if spec is None or spec.loader is None:
        raise RuntimeError(f"Cannot import X86 bus from {bus_file}")

    module = importlib.util.module_from_spec(spec)

    # X87B root-cause fix:
    # Python 3.13 dataclasses expect cls.__module__ to exist in sys.modules
    # while decorators execute during exec_module().
    sys.modules[module_name] = module

    spec.loader.exec_module(module)
    return module.AgentBus


class WorkerDispatcher:
    def __init__(self) -> None:
        STATE_DIR.mkdir(parents=True, exist_ok=True)
        REPORT_DIR.mkdir(parents=True, exist_ok=True)
        AgentBus = load_x86_bus_class()
        self.bus = AgentBus()
        self.state = load_json(STATE_FILE, {
            "release": "X87B_WORKER_DISPATCHER_IMPORTFIX",
            "created_at": utc_stamp(),
            "cycles": 0,
            "processed_total": 0,
            "last_cycle": None,
            "last_error": None,
        })

    def save(self) -> None:
        safe_write_json(STATE_FILE, self.state)

    def handle_message(self, msg: Dict[str, Any]) -> Dict[str, Any]:
        to_role = str(msg.get("to_role", "")).upper()
        task = str(msg.get("task", ""))
        payload = msg.get("payload") if isinstance(msg.get("payload"), dict) else {}

        result = {
            "message_id": msg.get("id"),
            "role": to_role,
            "task": task,
            "input_payload": payload,
            "handled_at": utc_stamp(),
            "ok": True,
            "summary": "",
            "routed_to": None,
        }

        if to_role == "DEBUG":
            result["summary"] = "DEBUG worker consumed evidence and confirmed root-cause-first handoff."
            result["output_payload"] = {
                "root_cause": payload.get("root_cause") or "pending_real_log_review",
                "evidence": payload.get("evidence") or payload,
                "source_message": msg.get("id"),
            }
        elif to_role == "PATCHER":
            result["summary"] = "PATCHER worker consumed confirmed cause/artifact context."
            result["output_payload"] = {
                "artifact": payload.get("artifact") or r"D:\openwebui\_patch_inbox",
                "patch_rule": "target confirmed cause only",
                "source_message": msg.get("id"),
            }
        elif to_role == "VERIFY":
            result["summary"] = "VERIFY worker consumed artifact and produced verify handoff."
            result["output_payload"] = {
                "verify": payload.get("verify") or "pending_x75_check",
                "source_message": msg.get("id"),
            }
        elif to_role == "REPORT":
            result["summary"] = "REPORT worker consumed final verification payload."
            result["output_payload"] = {
                "final": payload,
                "source_message": msg.get("id"),
            }
        else:
            result["ok"] = False
            result["summary"] = f"No worker handler for role {to_role}"
            result["output_payload"] = {"source_message": msg.get("id")}

        next_role = ROUTE_NEXT.get(to_role)
        if result["ok"] and next_role:
            sent = self.bus.send(to_role, next_role, task, result["output_payload"])
            result["routed_to"] = next_role
            result["routed_message_id"] = sent.id

        return result

    def cycle(self, roles: Optional[List[str]] = None, limit_per_role: int = 10) -> Dict[str, Any]:
        roles = [r.upper() for r in (roles or ["DEBUG", "PATCHER", "VERIFY", "REPORT"])]
        processed: List[Dict[str, Any]] = []

        for role in roles:
            consumed = self.bus.consume(role, limit=limit_per_role)
            for msg in consumed:
                processed.append(self.handle_message(msg))

        self.state["cycles"] = int(self.state.get("cycles", 0)) + 1
        self.state["processed_total"] = int(self.state.get("processed_total", 0)) + len(processed)
        self.state["last_cycle"] = {
            "ts": utc_stamp(),
            "roles": roles,
            "processed": len(processed),
        }
        self.state["last_error"] = None
        self.save()

        report = {
            "ok": True,
            "release": "X87B_WORKER_DISPATCHER_IMPORTFIX",
            "processed": processed,
            "processed_count": len(processed),
            "state_file": str(STATE_FILE),
        }
        out = REPORT_DIR / f"x87b_worker_dispatcher_report_{now_ms()}.json"
        safe_write_json(out, report)
        report["report_file"] = str(out)
        return report

    def seed(self, task: str) -> Dict[str, Any]:
        msg = self.bus.send("SCOUT", "DEBUG", task, {
            "finding": "seeded by X87B dispatcher",
            "evidence": r"D:\openwebui\runtime\temp",
        })
        return {"ok": True, "seed_message": msg.id, "task": task}

    def demo(self, task: str) -> Dict[str, Any]:
        seed = self.seed(task)
        cycles = []
        for _ in range(4):
            cycles.append(self.cycle(limit_per_role=20))
        report = {
            "ok": True,
            "release": "X87B_WORKER_DISPATCHER_IMPORTFIX",
            "task": task,
            "seed": seed,
            "cycles": cycles,
            "state_file": str(STATE_FILE),
        }
        out = REPORT_DIR / f"x87b_worker_dispatcher_demo_{now_ms()}.json"
        safe_write_json(out, report)
        report["report_file"] = str(out)
        return report

    def status(self) -> Dict[str, Any]:
        return {
            "ok": True,
            "release": "X87B_WORKER_DISPATCHER_IMPORTFIX",
            "state_file": str(STATE_FILE),
            "state": self.state,
            "x86_messages_dir": str(MESSAGES_DIR),
            "x86_exists": (CORE_DIR / "x86_agent_bus.py").exists(),
            "import_fix": "sys.modules[module_name] registered before exec_module",
        }


def cli() -> int:
    parser = argparse.ArgumentParser(description="X87B autonomous worker dispatcher importfix")
    sub = parser.add_subparsers(dest="cmd")

    p_seed = sub.add_parser("seed")
    p_seed.add_argument("--task", default="x87b seeded task")

    p_cycle = sub.add_parser("cycle")
    p_cycle.add_argument("--roles", default="DEBUG,PATCHER,VERIFY,REPORT")
    p_cycle.add_argument("--limit", type=int, default=10)

    p_demo = sub.add_parser("demo")
    p_demo.add_argument("--task", default="x87b dispatcher smoke")

    sub.add_parser("status")

    args = parser.parse_args()
    dispatcher = WorkerDispatcher()

    if args.cmd == "seed":
        print(json.dumps(dispatcher.seed(args.task), ensure_ascii=False, indent=2))
        return 0

    if args.cmd == "cycle":
        roles = [x.strip() for x in args.roles.split(",") if x.strip()]
        print(json.dumps(dispatcher.cycle(roles=roles, limit_per_role=args.limit), ensure_ascii=False, indent=2))
        return 0

    if args.cmd == "demo":
        print(json.dumps(dispatcher.demo(args.task), ensure_ascii=False, indent=2))
        return 0

    if args.cmd == "status":
        print(json.dumps(dispatcher.status(), ensure_ascii=False, indent=2))
        return 0

    parser.print_help()
    return 2


if __name__ == "__main__":
    raise SystemExit(cli())
