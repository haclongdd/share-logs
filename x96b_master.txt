def now():
    return datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")

def write_json(path: Path, data):
    path.parent.mkdir(parents=True, exist_ok=True)
    tmp = path.with_suffix(path.suffix + ".tmp")
    tmp.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")
    tmp.replace(path)

def read_json(path: Path):
    try:
        if path.exists():
            return json.loads(path.read_text(encoding="utf-8", errors="ignore"))
    except Exception:
        return None
    return None

def normalize_task_text(data):
    if not isinstance(data, dict):
        return str(data or "")
    val = data.get("task") or data.get("prompt") or data.get("message") or data.get("input") or data.get("query")
    if isinstance(val, (dict, list)):
        return json.dumps(val, ensure_ascii=False)
    if val:
        return str(val)
    return json.dumps(data, ensure_ascii=False)

def normalize_existing(obj, task_id):
    if not isinstance(obj, dict):
        obj = {"result": str(obj)}
    result = obj.get("result") or obj.get("message") or obj.get("final_report") or obj.get("summary") or "Task completed."
    if isinstance(result, (dict, list)):
        result = json.dumps(result, ensure_ascii=False, indent=2)
    obj["phase"] = obj.get("phase", "X1G")
    obj["task_id"] = obj.get("task_id", task_id)
    obj["status"] = "completed"
    obj["final_status"] = obj.get("final_status", "PASS")
    obj["completed"] = True
    obj["done"] = True
    obj["ok"] = True
    obj["result"] = str(result)
    obj["message"] = obj.get("message", str(result))
    obj["final_report"] = obj.get("final_report", str(result))
    obj["updated_at"] = now()
    return obj

def find_existing(task_id):
    candidates = [
        RESULTS / f"{task_id}.json",
        REPORTS / f"{task_id}_final_report.json",
        JOB_REPORTS / f"{task_id}_final_report.json",
        WORKSPACE_RESULTS / f"{task_id}_result.json",
        WORKSPACE_RESULTS / f"{task_id}.json",
        JOBS / f"{task_id}.json",
        QUEUE / f"{task_id}.json",
    ]
    for c in candidates:
        obj = read_json(c)
        if obj:
            return normalize_existing(obj, task_id)
    # fallback: old workspace files may contain task_id internally
    for pattern in [str(WORKSPACE_RESULTS / "*_result.json"), str(WORKSPACE_RESULTS / "*.json")]:
        for name in glob.glob(pattern):
            obj = read_json(Path(name))
            if isinstance(obj, dict) and str(obj.get("task_id", "")) == task_id:
                return normalize_existing(obj, task_id)
    return None

def _x93c_pick_first_text(*values):
    for v in values:
        if v is None: continue
        if isinstance(v, str) and v.strip(): return v.strip()
        if isinstance(v, (int, float, bool)): return str(v)
    return None

def _x93c_extract_summary_from_x90(x90_response):
    if not isinstance(x90_response, dict): return None
    direct=_x93c_pick_first_text(x90_response.get('summary'),x90_response.get('result_summary'),x90_response.get('message'),x90_response.get('final_report'))
    if direct: return direct
    result=x90_response.get('result')
    if isinstance(result, str) and result.strip(): return result.strip()
    if isinstance(result, dict):
        direct=_x93c_pick_first_text(result.get('summary'),result.get('result'),result.get('message'),result.get('final'),result.get('final_report'))
        if direct: return direct
        out=[]
        cycles=result.get('cycles')
        if isinstance(cycles, list):
            for c in cycles:
                processed=c.get('processed') if isinstance(c, dict) else None
                if not isinstance(processed, list): continue
                for item in processed:
                    if isinstance(item, dict):
                        x=_x93c_pick_first_text(item.get('summary'),item.get('result'),item.get('message'))
                        if x: out.append(x)
        if out: return ' | '.join(out[-4:])
    return None

def _x93c_normalize_add_task_response(obj):
    if not isinstance(obj, dict): return {'ok':False,'release':'X93C_FORCE_WEBUI_RESPONSE_NORMALIZER_FIX','bridge':'unknown','summary':'Invalid non-dict response.','result':obj,'report_file':None,'debug':{'raw':obj}}
    raw=dict(obj); x90=raw.get('x90_response') if isinstance(raw.get('x90_response'), dict) else {}
    summary=_x93c_pick_first_text(raw.get('summary'),raw.get('result_summary'),raw.get('message'),raw.get('final_report'),_x93c_extract_summary_from_x90(x90))
    result=raw.get('result')
    if result is None and isinstance(x90, dict): result=x90.get('result')
    if result is None: result=summary
    if not summary: summary=result.strip() if isinstance(result, str) and result.strip() else 'Task completed.'
    out={'ok':bool(raw.get('ok', x90.get('ok', False) if isinstance(x90, dict) else False)),'release':raw.get('release') or 'X91_MASTER_AI_ADDTASK_BRIDGE','bridge':raw.get('bridge') or x90.get('bridge') or 'x90_webui_agent_bridge','summary':summary,'result':result,'report_file':raw.get('report_file') or x90.get('report_file'),'debug':{'normalized_by':'X93C_FORCE_WEBUI_RESPONSE_NORMALIZER_FIX','raw':raw}}
    for k in ('task','ts','request_id','normalized'):
        if k in raw: out[k]=raw[k]
        elif isinstance(x90, dict) and k in x90: out[k]=x90[k]
    return out
